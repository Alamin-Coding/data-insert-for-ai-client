 [math]\Rightarrow[/math]
 [math]\left[\begin{array}{rr}1+0 & -1-3 \\0+0 & 0+9\end{array}\right][/math]
 [math]=\left[\begin{array}{rr}1 & -4 \\0 & 9\end{array}\right][/math]
 [math]\therefore P^{2}-2 I[/math]
 [math]=\left[\begin{array}{rr}1 & -4 \\0 & .9\end{array}\right][/math]
 [math]-2\left[\begin{array}{rr}1 & 0 \\0 & 1\end{array}\right][/math]
 [math]=\left[\begin{array}{rr}-1 & -4 \\0 & 7\end{array}\right][/math]
