 [math]\begin{array}{l}\Rightarrow y=x+1 \\ \quad y=m_{2} x+c \\m_{2}=\tan \mathrm{\theta}_{2}=+1 \\\theta_{2}=\tan ^{-1}(1)=45^{\circ} \\y+3=0 \\\therefore y=-3\end{array}[/math]
